UPDATER = """\
\"\"\"Auto README Updater — fetches live GitHub data and injects it into README.md\"\"\"
 
import os, re, requests
from datetime import datetime, timezone
 
USERNAME = os.environ.get("GITHUB_USERNAME", "AhmedRaza-2")
TOKEN    = os.environ.get("GITHUB_TOKEN", "")
HEADERS  = {"Authorization": f"Bearer {TOKEN}", "Accept": "application/vnd.github+json"}
 
LANG_COLORS = {
    "Python": "3776AB", "JavaScript": "F7DF1E", "TypeScript": "3178C6",
    "HTML": "E34F26", "CSS": "1572B6", "Java": "ED8B00",
    "C++": "00599C", "C#": "512BD4", "Dart": "0175C2",
    "Shell": "89e051", "Jupyter Notebook": "DA5B0B",
}
 
def fetch_repos():
    r = requests.get(f"https://api.github.com/users/{USERNAME}/repos",
                     headers=HEADERS, params={"per_page": 100, "sort": "updated"})
    r.raise_for_status()
    repos = [x for x in r.json() if not x.get("fork")]
    repos.sort(key=lambda x: x.get("stargazers_count", 0), reverse=True)
    return repos
 
def fetch_user():
    r = requests.get(f"https://api.github.com/users/{USERNAME}", headers=HEADERS)
    r.raise_for_status()
    return r.json()
 
def repo_table(repos, n=6):
    rows = []
    for repo in repos[:n]:
        name  = repo["name"]
        desc  = (repo.get("description") or "No description.")[:72]
        stars = repo.get("stargazers_count", 0)
        forks = repo.get("forks_count", 0)
        lang  = repo.get("language") or "N/A"
        url   = repo["html_url"]
        color = LANG_COLORS.get(lang, "8B949E")
        badge = f'<img src="https://img.shields.io/badge/{lang.replace(" ", "%20").replace("#", "%23")}-{color}?style=flat-square"/>'
        try:
            updated = datetime.fromisoformat(repo["updated_at"].replace("Z","+00:00")).strftime("%b %Y")
        except Exception:
            updated = "N/A"
        rows.append(f"| [**{name}**]({url}) | {desc} | ⭐ {stars} | 🍴 {forks} | {badge} | {updated} |")
    header = "| Repository | Description | Stars | Forks | Language | Updated |\\n|------------|-------------|:-----:|:-----:|----------|---------|"
    return header + "\\n" + "\\n".join(rows)
 
def stats_row(user, repos):
    stars = sum(r.get("stargazers_count", 0) for r in repos)
    forks = sum(r.get("forks_count", 0) for r in repos)
    synced = datetime.now(timezone.utc).strftime("%d %b %Y, %H:%M UTC")
    return (
        "| ⭐ Total Stars | 🍴 Total Forks | 👥 Followers | 📦 Public Repos | 🕒 Last Synced |\\n"
        "|:---:|:---:|:---:|:---:|:---:|\\n"
        f"| **{stars}** | **{forks}** | **{user['followers']}** | **{user['public_repos']}** | `{synced}` |"
    )
 
def update(readme_path="README.md"):
    repos = fetch_repos()
    user  = fetch_user()
    with open(readme_path, "r", encoding="utf-8") as f:
        content = f.read()
    content = re.sub(r"<!-- REPOS-START -->.*?<!-- REPOS-END -->",
                     f"<!-- REPOS-START -->\\n{repo_table(repos)}\\n<!-- REPOS-END -->",
                     content, flags=re.DOTALL)
    content = re.sub(r"<!-- LIVE-STATS-START -->.*?<!-- LIVE-STATS-END -->",
                     f"<!-- LIVE-STATS-START -->\\n{stats_row(user, repos)}\\n<!-- LIVE-STATS-END -->",
                     content, flags=re.DOTALL)
    with open(readme_path, "w", encoding="utf-8") as f:
        f.write(content)
    print(f"✅ README updated — ⭐ {sum(r.get('stargazers_count',0) for r in repos)} stars across {len(repos)} repos")
 
if __name__ == "__main__":
    update()
"""
